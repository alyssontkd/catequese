CREATE 
    ALGORITHM = UNDEFINED 
    DEFINER = `root`@`localhost` 
    SQL SECURITY DEFINER
VIEW `bdcatequese`.`auth` AS
    (SELECT 
        `bdcatequese`.`login`.`id_usuario` AS `id_usuario`,
        `bdcatequese`.`perfil`.`id_perfil` AS `id_perfil`,
        `bdcatequese`.`email`.`em_email` AS `em_email`,
        `bdcatequese`.`login`.`pw_senha` AS `pw_senha`,
        `bdcatequese`.`usuario`.`nm_usuario` AS `nm_usuario`,
        1 AS `id_contrato`
    FROM
        (((`bdcatequese`.`usuario`
        JOIN `bdcatequese`.`login` ON ((`bdcatequese`.`login`.`id_usuario` = `bdcatequese`.`usuario`.`id_usuario`)))
        JOIN `bdcatequese`.`email` ON ((`bdcatequese`.`email`.`id_email` = `bdcatequese`.`login`.`id_email`)))
        JOIN `bdcatequese`.`perfil` ON ((`bdcatequese`.`perfil`.`id_perfil` = `bdcatequese`.`login`.`id_perfil`))));
